% main_base_original.tex contains the original \documentclass line. This wrapper

\newif\ifInsertMeshGraphNetSlides

  \def\ModelSlidesCurrentTitle{#1}%
  \def\ModelSlidesFNOTitle{Fourier Neural Operator (FNO)}%
  \def\ModelSlidesMGNTitle{MeshGraphNet}%
  \ifx\ModelSlidesCurrentTitle\ModelSlidesFNOTitle
    \global\InsertFNOSlidestrue
  \fi
  \ifx\ModelSlidesCurrentTitle\ModelSlidesMGNTitle
    \global\InsertMeshGraphNetSlidestrue
  \fi
}
\apptocmd{\endframe}{%
  \ifInsertFNOSlides
    \global\InsertFNOSlidesfalse
    \input{fno_slides}%
  \fi
  \ifInsertMeshGraphNetSlides
    \global\InsertMeshGraphNetSlidesfalse
    \input{meshgraphnet_slides}%
  \fi
}{}{}
\makeatother
