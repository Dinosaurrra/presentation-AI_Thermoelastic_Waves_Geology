% main.tex -- wrapper for the base Beamer deck.
% The original deck body is stored in main_base.tex.
% Additional FNO slides are inserted before the Calculation Results section.

\AtBeginDocument{%
  \let\FNOOriginalSection\section
  \renewcommand{\section}[1]{%
    \def\FNOCurrentSection{#1}%
    \def\FNOCalculationResults{Calculation Results}%
    \ifx\FNOCurrentSection\FNOCalculationResults
      \input{fno_slides}%
    \fi
    \FNOOriginalSection{#1}%
  }%
}

\input{main_base.tex}
