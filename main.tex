% main.tex -- wrapper for the base Beamer deck.
% The original deck body is stored in main_base.tex.
% Additional model-detail slides are inserted before the Calculation Results section.

\documentclass[aspectratio=169,10pt]{beamer}

% main_base.tex contains the original \documentclass line. The wrapper has
% already loaded the class, so that line is ignored when the base file is read.
\renewcommand{\documentclass}[2][]{}

\AtBeginDocument{%
  \let\ModelSlidesOriginalSection\section
  \renewcommand{\section}[1]{%
    \def\ModelSlidesCurrentSection{#1}%
    \def\ModelSlidesCalculationResults{Calculation Results}%
    \ifx\ModelSlidesCurrentSection\ModelSlidesCalculationResults
      \input{fno_slides}%
      \input{meshgraphnet_slides}%
    \fi
    \ModelSlidesOriginalSection{#1}%
  }%
}

\input{main_base.tex}
